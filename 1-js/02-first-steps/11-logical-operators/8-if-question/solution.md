คำตอบคือ "if" ตัวแรกและตัวที่สามจะทำงาน

คำอธิบาย:

```js run
// คำสั่งนี้ทำงาน
// ผลลัพธ์คือ -1 || 0 = -1 เป็นค่า truthy
if (-1 || 0) alert( 'first' );

// คำสั่งนี้ไม่ทำงาน
// -1 && 0 = 0 เป็นค่า falsy
if (-1 && 0) alert( 'second' );

// คำสั่งนี้ทำงาน
// ตัวดำเนินการ && จะทำงานก่อน || เสมอ
// ดังนั้น -1 && 1 ทำงานเสร็จ ก็จะทำงานส่วนที่เหลือต่อ
// null || -1 && 1  จะเป็น  null || 1  จะเป็น  1
if (null || -1 && 1) alert( 'third' );
```
